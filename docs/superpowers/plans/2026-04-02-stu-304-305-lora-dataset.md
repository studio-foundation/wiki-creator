**Tech Stack:** Python 3, standard library only (json, re, argparse, pathlib). Reuses `build_prompt` from `scripts/generate_wiki_pages.py`.

```python
# wiki_creator/wikitext_cleaner.py
"""Convert fandom wikitext to clean Markdown."""

from __future__ import annotations

import re


def clean_wikitext(text: str) -> str:
    """Transform wikitext into pipeline-compatible Markdown.

    Steps applied in order:
    1. Strip metadata (categories, galleries, templates, references, file links)
    2. Convert wiki syntax to Markdown (bold, italic, links)
    3. Remove empty sections (header with no content before next header)
    4. Normalize whitespace
    """
    # --- Strip metadata ---
    # Categories
    text = re.sub(r"\[\[Category:[^\]]*\]\]\n?", "", text)
    # Gallery blocks
    text = re.sub(r"<gallery[^>]*>.*?</gallery>\n?", "", text, flags=re.DOTALL)
    # Templates {{...}}
    text = re.sub(r"\{\{[^}]*\}\}", "", text)
    # <references/> or <references>...</references>
    text = re.sub(r"<references\s*/?>", "", text)
    text = re.sub(r"<references>.*?</references>", "", text, flags=re.DOTALL)
    # File/Image links
    text = re.sub(r"\[\[(?:File|Image):[^\]]*\]\]\n?", "", text)

    # --- Convert wiki syntax ---
    # Bold: '''text''' -> **text**
    text = re.sub(r"'''(.*?)'''", r"**\1**", text)
    # Italic: ''text'' -> *text*
    text = re.sub(r"''(.*?)''", r"*\1*", text)
    # Links with display text: [[Target|Display]] -> Display
    text = re.sub(r"\[\[[^\]|]+\|([^\]]+)\]\]", r"\1", text)
    # Plain links: [[Target]] -> Target
    text = re.sub(r"\[\[([^\]]+)\]\]", r"\1", text)

    # --- Remove empty sections ---
    # A section is empty if its header is followed by nothing (or only whitespace)
    # before the next header of same or higher level, or end of string.
    text = re.sub(
        r"(?m)^(#{2,})\s+[^\n]+\n+(?=#{1,}\s|\Z)",
        "",
        text,
    )

    # --- Normalize whitespace ---
    text = re.sub(r"\n{3,}", "\n\n", text)
    text = text.strip()

    return text
```

```python
# EN→FR key mapping for infobox fields
_INFOBOX_KEY_MAP: dict[str, str] = {
    "name": "nom",
    "full name": "nom complet",
    "AKA": "alias",
    "alias": "alias",
    "allegiance": "affiliation",
    "status": "statut",
    "gender": "genre",
    "title": "titre",
    "role": "rôle",
    "occupation": "rôle",
    "location": "lieu",
    "eye color": "couleur des yeux",
    "hair color": "couleur des cheveux",
    "skin color": "couleur de peau",
}

# Keys to drop entirely (UI/media fields)
_INFOBOX_DROP_KEYS: set[str] = {
    "image", "caption", "affcollapse", "statcollapse",
    "appearances", "gallery",
}


def _strip_wiki_links(value: str) -> str:
    """Remove [[...]] wiki link markup from a string value."""
    value = re.sub(r"\[\[[^\]|]+\|([^\]]+)\]\]", r"\1", value)
    value = re.sub(r"\[\[([^\]]+)\]\]", r"\1", value)
    return value


def normalize_infobox_fields(fields: dict[str, str]) -> dict[str, str]:
    """Normalize fandom infobox fields to pipeline-compatible French keys."""
    result: dict[str, str] = {}
    for key, value in fields.items():
        if key.lower() in _INFOBOX_DROP_KEYS or key.lower().endswith("collapse"):
            continue
        new_key = _INFOBOX_KEY_MAP.get(key, key)
        new_value = _strip_wiki_links(str(value)) if isinstance(value, str) else str(value)
        result[new_key] = new_value
    return result
```

```python
#!/usr/bin/env python3
# scripts/export_lora_dataset.py
"""Export LoRA training dataset in Alpaca JSONL format.

Assembles data from:
- Fandom scrape JSONL (wikitext cleaned to Markdown)
- Pipeline wiki_pages.json outputs
- Gold examples JSONL

Splits into train/eval based on entity importance.
"""
from __future__ import annotations

import argparse
import json
import sys
from pathlib import Path

# Allow imports from project root
sys.path.insert(0, str(Path(__file__).resolve().parent.parent))

from wiki_creator.wikitext_cleaner import clean_wikitext, normalize_infobox_fields

VALID_ENTITY_TYPES = {"PERSON", "PLACE", "ORG", "OTHER"}
VALID_IMPORTANCES = {"principal", "secondary", "figurant"}
MIN_CONTENT_LENGTH = 50


def infer_importance(content: str) -> str:
    """Infer entity importance from content length."""
    length = len(content)
    if length > 2000:
        return "principal"
    if length > 500:
        return "secondary"
    return "figurant"


def validate_output(output: dict) -> bool:
    """Validate an output dict meets LoRA dataset requirements."""
    if not output.get("title"):
        return False
    if not output.get("content"):
        return False
    if output.get("entity_type") not in VALID_ENTITY_TYPES:
        return False
    if output.get("importance") not in VALID_IMPORTANCES:
        return False
    # Non-figurants must have at least one ## header
    if output["importance"] != "figurant" and "## " not in output["content"]:
        return False
    return True


def build_fandom_instruction(title: str, entity_type: str, importance: str) -> str:
    """Build a synthetic instruction prompt for a fandom entry.

    Uses a simplified version of the build_prompt format since fandom entries
    lack context_by_chapter and relationships data.
    """
    return (
        f'This is a fictional world.\n\n'
        f'You are writing a wiki page for a fictional novel.\n'
        f'Output ONLY a valid JSON object. No markdown fences. No explanation.\n\n'
        f'---\n\n'
        f'Entity to write:\n'
        f'  Name: {title}\n'
        f'  Type: {entity_type}\n'
        f'  Importance: {importance}\n\n'
        f'---\n\n'
        f'Output this JSON object:\n'
        f'{{\n'
        f'  "title": "{title}",\n'
        f'  "importance": "{importance}",\n'
        f'  "entity_type": "{entity_type}",\n'
        f'  "infobox_fields": {{}},\n'
        f'  "content": "<Markdown string>"\n'
        f'}}\n\n'
        f'Output ONLY the JSON. Nothing before, nothing after.'
    )


def process_fandom_entry(
    entry: dict,
    known_entities: dict[str, dict],
) -> dict | None:
    """Process a single fandom JSONL entry into an Alpaca-format dict.

    Returns None if the entry should be rejected (content too short).
    """
    title = entry["page_title"]
    entity_type = entry.get("entity_type", "OTHER")
    raw_content = entry.get("content", "")
    raw_infobox = entry.get("infobox_fields", {})

    # Clean content
    content = clean_wikitext(raw_content)
    if len(content) < MIN_CONTENT_LENGTH:
        return None

    # Normalize infobox
    infobox = normalize_infobox_fields(raw_infobox)

    # Determine importance
    known = known_entities.get(title, {})
    importance = known.get("importance") or infer_importance(content)

    output = {
        "title": title,
        "importance": importance,
        "entity_type": entity_type,
        "infobox_fields": infobox,
        "content": content,
    }

    if not validate_output(output):
        return None

    instruction = build_fandom_instruction(title, entity_type, importance)

    return {
        "instruction": instruction,
        "input": "",
        "output": json.dumps(output, ensure_ascii=False),
    }


def load_known_entities(batch_dir: Path) -> dict[str, dict]:
    """Load entity metadata from batch files for importance lookup."""
    known: dict[str, dict] = {}
    for batch_file in sorted(batch_dir.glob("batch_*.json")):
        with open(batch_file) as f:
            batch = json.load(f)
        for entity in batch.get("entities", []):
            name = entity.get("canonical_name", "")
            if name:
                known[name] = {
                    "importance": entity.get("importance", "figurant"),
                    "type": entity.get("type", "OTHER"),
                }
    return known


def load_pipeline_outputs(wiki_pages_path: Path, batch_dir: Path) -> list[dict]:
    """Load validated pipeline outputs and reconstruct instruction via build_prompt."""
    if not wiki_pages_path.exists():
        return []

    # Import build_prompt from generate_wiki_pages
    scripts_dir = Path(__file__).resolve().parent
    gen_spec = __import__("importlib").util.spec_from_file_location(
        "generate_wiki_pages", scripts_dir / "generate_wiki_pages.py"
    )
    gen_mod = __import__("importlib").util.module_from_spec(gen_spec)
    gen_spec.loader.exec_module(gen_mod)

    with open(wiki_pages_path) as f:
        data = json.load(f)

    # Build entity lookup from batch files
    entity_lookup: dict[str, dict] = {}
    for batch_file in sorted(batch_dir.glob("batch_*.json")):
        with open(batch_file) as f:
            batch = json.load(f)
        for entity in batch.get("entities", []):
            entity_lookup[entity["canonical_name"]] = entity

    results = []
    for page in data.get("pages", []):
        if page.get("_failed") or page.get("_spoiler_rejected"):
            continue
        title = page.get("title", "")
        if title not in entity_lookup:
            continue

        entity = entity_lookup[title]
        etype = entity.get("type", "PERSON")
        sections = list(
            gen_mod.SECTION_DEFINITIONS.get(etype, gen_mod.SECTION_DEFINITIONS["PERSON"]).keys()
        )
        instruction = gen_mod.build_prompt(entity, "Throne of Glass", sections)

        output = {
            "title": title,
            "importance": page.get("importance", "figurant"),
            "entity_type": page.get("entity_type", etype),
            "infobox_fields": page.get("infobox_fields", {}),
            "content": page.get("content", ""),
        }
        if validate_output(output):
            results.append({
                "instruction": instruction,
                "input": "",
                "output": json.dumps(output, ensure_ascii=False),
            })
    return results


def main() -> None:
    parser = argparse.ArgumentParser(description="Export LoRA dataset in Alpaca JSONL format")
    parser.add_argument("--fandom-jsonl", type=Path, required=True, help="Path to fandom JSONL")
    parser.add_argument(
        "--batch-dir",
        type=Path,
        default=None,
        help="Path to wiki_inputs/<slug>/ with batch_*.json files (for importance lookup)",
    )
    parser.add_argument(
        "--wiki-pages",
        type=Path,
        default=None,
        help="Path to wiki_pages.json (pipeline outputs)",
    )
    parser.add_argument(
        "--gold-jsonl",
        type=Path,
        default=None,
        help="Path to gold_examples.jsonl",
    )
    parser.add_argument("--output-dir", type=Path, required=True, help="Output directory")
    args = parser.parse_args()

    args.output_dir.mkdir(parents=True, exist_ok=True)

    # Load known entities from batch files
    known_entities: dict[str, dict] = {}
    if args.batch_dir and args.batch_dir.exists():
        known_entities = load_known_entities(args.batch_dir)

    # --- Process fandom entries ---
    train_examples: list[dict] = []
    eval_examples: list[dict] = []
    report = {"fandom_total": 0, "fandom_accepted": 0, "fandom_rejected": 0,
              "pipeline_accepted": 0, "gold_accepted": 0,
              "rejected_pages": [], "by_type": {}, "by_importance": {}}

    with open(args.fandom_jsonl) as f:
        for line in f:
            entry = json.loads(line)
            report["fandom_total"] += 1
            result = process_fandom_entry(entry, known_entities)
            if result is None:
                report["fandom_rejected"] += 1
                report["rejected_pages"].append(entry.get("page_title", "unknown"))
                continue
            report["fandom_accepted"] += 1
            output = json.loads(result["output"])
            etype = output["entity_type"]
            importance = output["importance"]
            report["by_type"][etype] = report["by_type"].get(etype, 0) + 1
            report["by_importance"][importance] = report["by_importance"].get(importance, 0) + 1

            if importance == "principal":
                eval_examples.append(result)
            else:
                train_examples.append(result)

    # --- Process pipeline outputs ---
    if args.wiki_pages and args.batch_dir:
        pipeline_results = load_pipeline_outputs(args.wiki_pages, args.batch_dir)
        train_examples.extend(pipeline_results)
        report["pipeline_accepted"] = len(pipeline_results)

    # --- Load gold examples ---
    if args.gold_jsonl and args.gold_jsonl.exists():
        with open(args.gold_jsonl) as f:
            for line in f:
                example = json.loads(line)
                train_examples.append(example)
                report["gold_accepted"] += 1

    # --- Write outputs ---
    train_path = args.output_dir / "train.jsonl"
    eval_path = args.output_dir / "eval.jsonl"
    report_path = args.output_dir / "export_report.json"

    with open(train_path, "w") as f:
        for ex in train_examples:

    with open(eval_path, "w") as f:
        for ex in eval_examples:
            f.write(json.dumps(ex, ensure_ascii=False) + "\n")

    report["train_count"] = len(train_examples)
    report["eval_count"] = len(eval_examples)

    with open(report_path, "w") as f:
        json.dump(report, f, ensure_ascii=False, indent=2)

    print(f"Train: {len(train_examples)} examples -> {train_path}")
    print(f"Eval:  {len(eval_examples)} examples -> {eval_path}")
    print(f"Report: {report_path}")

    main()
```
